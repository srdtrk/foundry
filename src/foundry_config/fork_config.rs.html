10/// Fork-scoped config for tests and scripts.
11#[derive(Clone, Debug, PartialEq, Serialize, Deserialize, Default)]
12pub struct ForkConfigs(pub HashMap<String, ForkChainConfig>);
13
14impl ForkConfigs {
15    /// Resolve environment variables in all fork config fields
16    pub fn resolve_env_vars(&mut self) -> Result<(), ExtractConfigError> {
17        for (name, fork_config) in &mut self.0 {
18            // Take temporary ownership of the config, so that it can be consumed.
19            let config = std::mem::take(fork_config);
20
21            // Resolve the env vars and place it back into the map.
22            *fork_config = config.resolved().map_err(|e| {
23                let msg = if !e.var.is_empty() {
24                    format!("environment variable `{}` not found", e.var)
25                } else {
26                    e.to_string()
27                };
28                ExtractConfigError::new(figment::Error::from(format!(
29                    "Failed to resolve fork config [forks.{name}]: {msg}"
30                )))
31            })?;
32        }
33
34        Ok(())
35    }
36}

46/// Fork-scoped config for tests and scripts.
47#[derive(Clone, Debug, PartialEq, Serialize, Deserialize, Default)]
48pub struct ForkChainConfig {
49    // Optional RPC endpoint for the fork.
50    //
51    // If uninformed, it will attempt to load one from `[rpc_endpoints]` with a matching alias
52    // for the name of the forked chain.
53    pub rpc_endpoint: Option<RpcEndpoint>,
54    // Any arbitrary key-value pair of variables.
55    pub vars: HashMap<String, toml::Value>,
56}
57
58impl ForkChainConfig {
59    /// Resolves environment variables in the fork configuration.
60    /// Returns a new ForkConfig with all environment variables resolved.
61    pub fn resolved(self) -> Result<Self, UnresolvedEnvVarError> {
62        let mut resolved_vars = HashMap::new();
63        for (key, value) in self.vars {
64            resolved_vars.insert(key, resolve_toml_value(value)?);
65        }
66
67        Ok(Self { rpc_endpoint: self.rpc_endpoint, vars: resolved_vars })
68    }
69}
70
71/// Recursively resolves environment variables in a toml::Value
72fn resolve_toml_value(value: toml::Value) -> Result<toml::Value, UnresolvedEnvVarError> {
73    match value {
74        toml::Value::String(s) => {
75            // Check if the string contains environment variable placeholders
76            if RE_PLACEHOLDER.is_match(&s) {
77                // Resolve the environment variables
78                let resolved = interpolate(&s)?;
79                Ok(toml::Value::String(resolved))
80            } else {
81                Ok(toml::Value::String(s))
82            }
83        }
84        toml::Value::Array(arr) => {
85            // Recursively resolve each element in the array
86            let resolved_arr: Result<Vec<_>, _> = arr.into_iter().map(resolve_toml_value).collect();
87            Ok(toml::Value::Array(resolved_arr?))
88        }
89        toml::Value::Table(table) => {
90            // Recursively resolve each value in the table
91            let mut resolved_table = toml::map::Map::new();
92            for (k, v) in table {
93                resolved_table.insert(k, resolve_toml_value(v)?);
94            }
95            Ok(toml::Value::Table(resolved_table))
96        }
97        // Other types don't need resolution
98        other => Ok(other),
99    }
100}
